using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication6
{
    class Program
    {
        static void Main(string[] args)
        {
            int a, b, c, delta;
            double kok1, kok2;
            Console.Write("Birinci kat sayýyý sayýyý gir:   ");
            a = Convert.ToInt32(Console.ReadLine());
            Console.WriteLine("Ýkinci sayýyý gir: ");
            b = Convert.ToInt32(Console.ReadLine());
            Console.WriteLine("Üçüncü katsayýyý gir: ");
            c = Convert.ToInt32(Console.ReadLine());
            delta = b * b - 4 * a * c;
            if (delta < 0)
                Console.Write("Tam kok yok");
            else if (delta == 0)
            {
                kok1 = -b / 2 * a;
                Console.Write("Tek kok var = " + kok1);
            }
            else
            {
                kok1 = (-b + Math.Sqrt(delta)) / 2 * a;
                kok2 = (-b - Math.Sqrt(delta)) / 2 * a;
                Console.Write("Ýki kok var , 1.kok: {0}, 2.kok: {1}", kok1, kok2);
            }
            Console.ReadKey();
        }
    }
}
